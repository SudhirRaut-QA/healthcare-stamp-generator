# GitHub Setup Commands
# Replace 'your-username' with your actual GitHub username

# Add the remote origin (replace 'your-username' with your GitHub username)
git remote add origin https://github.com/your-username/healthcare-stamp-generator.git

# Rename the default branch to main (GitHub's new standard)
git branch -M main

# Push the code to GitHub
git push -u origin main

# Push all branches
git push -u origin develop
git push -u origin feature/enhanced-stamps

# Verify the remote connection
git remote -v
